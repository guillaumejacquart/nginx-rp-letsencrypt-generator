server {
        listen 80;
        return 301 https://$host$request_uri;
}

server {
        listen 443 ssl;
        server_name test.dynalias.org;

        ssl_certificate /etc/letsencrypt/live/test.dynalias.org/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/test.dynalias.org/privkey.pem;

        access_log /var/log/nginx/test.dynalias.org.log combined;

        location /.well-known {
                alias /var/www/test.dynalias.org/.well-known;
        }

        location / {
                proxy_pass http://192.168.0.1:80;
                proxy_set_header        Host    $host;
                proxy_set_header        X-Forwarded-For $remote_addr;
        }
}
